\Function{\textsc{annotatetx}$_\Sidechain$}{$\chainsc, \chainmc$}
    \Let{\vec\tx}{\emptyset}
    \label{alg.sidechain.interspersing}
    \For{\text{each time slot }$t$}
        \Let{\vec\tx'}{\epsilon}
        \If{$\chainsc$ has a block generated at slot $t$}
            \Let{B}{\text{the block in }\chainsc\text{ generated at }t}
            \For{$\tx \in B$}
                \Let{\vec\tx'}{\vec\tx' \concat \tx}
            \EndFor
        \EndIf
        \If{$\chainmc$ has a block generated at slot $t$}
            \Let{B}{\text{the block in }\chainmc\text{ generated at }t}
            \For{$\tx \in B$}
                \Let{\vec\tx'}{\vec\tx' \concat \tx}
            \EndFor
        \EndIf
        \For{$\tx \in \vec\tx'$}
            \CommentLine{Mark the time of each $\tx$ in $\vec\tx'$}
            \Let{\tx\text{.}t}{t}
        \EndFor
        \Let{\vec\tx}{\vec\tx \concat \vec\tx'}
    \EndFor
    \Return{$\vec\tx$}
\EndFunction
